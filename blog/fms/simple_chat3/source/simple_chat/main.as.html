load( "chat.asc" );
load( "users.asc" );

// called when client attempts to enter app
application.onConnect = function( client )
{
    client.chat        = chat;
    client.users    = users;
    
    // get a unique user object for this client
    var user        = users.addUser( client );
    // accept client connection
    application.acceptConnection( client );
    // pass the user object back to the client
    client.call("setUser", null, user);
}

// called when client leaves app
application.onDisconnect = function( client )
{
    // remove the reference to the user
    users.removeUser( client );
}
