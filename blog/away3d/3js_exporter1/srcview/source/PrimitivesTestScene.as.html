package
{
    import away3d.cameras.TargetCamera3D;
    import away3d.containers.View3D;
    import away3d.events.ExporterEvent;
    import away3d.primitives.Cone;
    import away3d.primitives.Cube;
    import away3d.primitives.Cylinder;
    import away3d.primitives.GeodesicSphere;
    import away3d.primitives.GridPlane;
    import away3d.primitives.LineSegment;
    import away3d.primitives.Plane;
    import away3d.primitives.RegularPolygon;
    import away3d.primitives.Sphere;
    import away3d.primitives.Torus;
    import away3d.primitives.Trident;
    import away3d.primitives.WireCone;
    import away3d.primitives.WireCube;
    import away3d.primitives.WireCylinder;
    import away3d.primitives.WirePlane;
    import away3d.primitives.WireRegularPolygon;
    import away3d.primitives.WireSphere;
    import away3d.primitives.WireTorus;
    
    import components.ExportButton;
    
    import exporters.ThreeJSExporter;
    
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.KeyboardEvent;
    import flash.events.MouseEvent;
    import flash.geom.Vector3D;
    import flash.system.System;
    import flash.text.TextField;
    import flash.ui.Keyboard;
    
    [SWF("800", height="600", backgroundColor="0xFFFFFF", frameRate="30" )]
    public class PrimitivesTestScene extends Sprite
    {
        private var view:View3D;
        
        private var cam:TargetCamera3D;
        private var lastKey:uint;
        private var keyIsDown:Boolean = false;
        
        private static const KEY_A        :uint = 65;
        private static const KEY_D        :uint = 68;
        private static const KEY_S        :uint = 83;
        private static const KEY_W        :uint = 87;
        
        private var feedbackTf            :TextField;
        
        public function PrimitivesTestScene()
        {        
            // create a 3D-viewport
            view     = new View3D();
            view.x    = stage.stageWidth / 2;
            view.y    = stage.stageHeight / 2;
            addChild(view);
            
            var zpos    :int = -80;
            createPrimitives(0, 0, zpos);
            
            // create a basic camera
            cam         = new TargetCamera3D();
            cam.y        = 120;
            cam.z        = 240;
            view.camera = cam;
            
            stage.addEventListener(KeyboardEvent.KEY_DOWN, keyDown);
            stage.addEventListener(KeyboardEvent.KEY_UP, keyUp);
            addEventListener(Event.ENTER_FRAME, onEnterFrame);
            
            var exportBtn    :ExportButton = new ExportButton();
            exportBtn.addEventListener( MouseEvent.CLICK, clickExportBtnHandler );
            exportBtn.x     = 10;
            exportBtn.y     = 10;
            addChild(exportBtn);
            
            feedbackTf            = new TextField();
            feedbackTf.selectable = false;
            feedbackTf.width    = 200;
            feedbackTf.x        = 130;
            feedbackTf.y        = 10;
            addChild( feedbackTf );
        }
        
        private function createPrimitives(xpos:int, ypos:int, zpos:int):void
        {
            var plane:Plane = new Plane({width:200, height:50, pushback:true, x:xpos, y:ypos, z:zpos});            
            view.scene.addChild(plane);
            
            xpos                -= 73;
            ypos                += 10;
            
            var initXpos    :int = xpos;
            var initYpos    :int = ypos;
            
            var sphere:Sphere = new Sphere({radius:10, segmentsW:12, segmentsH:9, x:xpos, y:ypos, z:zpos});
            view.scene.addChild(sphere);
            
            xpos += 30;
            
            var wireSphere:WireSphere = new WireSphere({radius:10, segmentsW:12, segmentsH:9, x:xpos, y:ypos, z:zpos});
            view.scene.addChild(wireSphere);
            
            xpos += 30;
            
            var cube:Cube = new Cube({width:20, height:20, depth:20, x:xpos, y:ypos, z:zpos});
            cube.rotationX = 30;
            view.scene.addChild( cube );
            
            xpos += 30;
            
            var wireCube:WireCube = new WireCube({width:20, height:20, depth:20, x:xpos, y:ypos, z:zpos});
            wireCube.rotationY = 30;
            view.scene.addChild( wireCube );
            
            xpos += 30;
            
            var cone:Cone = new Cone({radius:10, height:20, segmentsW:6, segmentsH:4, openEnded:false, yUp:true, x:xpos, y:ypos, z:zpos});
            cone.rotationX = 45;
            cone.rotationY = 45;
            cone.rotationZ = 45;
            view.scene.addChild( cone );
            
            xpos += 30;
            
            var wireCone:WireCone = new WireCone({radius:10, height:20, segmentsW:6, segmentsH:4, yUp:true, x:xpos, y:ypos, z:zpos});
            wireCone.rotationZ = 30;
            view.scene.addChild( wireCone );
            
            xpos = initXpos;
            zpos += 60;
            
            var cylinder:Cylinder = new Cylinder({radius:10, height:20, segmentsW:6, segmentsH:4, openEnded:false, yUp:true, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( cylinder );
            
            xpos += 30;
            
            var wireCylinder:WireCylinder = new WireCylinder({radius:10, height:20, segmentsW:6, segmentsH:4, yUp:true, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( wireCylinder );
            
            xpos += 30;
            
            var lineSeg:LineSegment = new LineSegment({start:new Vector3D(xpos-10, ypos, zpos), end:new Vector3D(xpos+10, ypos, zpos)})
            view.scene.addChild( lineSeg );
            
            xpos += 30;
            
            var geoSphere:GeodesicSphere = new GeodesicSphere({radius:10, fractures:6, x:xpos, y:ypos, z:zpos});
            geoSphere.radius = 10;
            view.scene.addChild( geoSphere );
            
            xpos += 30;
            
            var gridPlane:GridPlane = new GridPlane({width:200, height:50, segmentsW:4, segmentsH:4, yUp:true, pushback:true, x:initXpos+73, y:initYpos-10, z:zpos});
            view.scene.addChild( gridPlane );
            
            var wirePlane:WirePlane = new WirePlane({width:200, height:50, segmentsW:4, segmentsH:4, yUp:true, pushback:true, x:initXpos+73, y:initYpos-10, z:zpos+60});
            view.scene.addChild( wirePlane );
            
            var regPoly:RegularPolygon = new RegularPolygon({radius:10, sides:8, subdivision:1, yUp:true, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( regPoly );
            
            xpos += 30;
            
            var wireRegPoly:WireRegularPolygon = new WireRegularPolygon({radius:10, sides:8, yUp:true, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( wireRegPoly );
        
            xpos += 30;    
            
            xpos = initXpos;
            zpos += 60;
            
            var torus:Torus    = new Torus({radius:10, tube:3, segmentsR:8, segmentsT:6, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( torus );
            
            xpos += 30;
            
            var wireTorus:WireTorus    = new WireTorus({radius:10, tube:3, segmentsR:8, segmentsT:6, x:xpos, y:ypos, z:zpos});
            view.scene.addChild( wireTorus );
            
            xpos += 30;
            
            var trident:Trident = new Trident(200, true);
            view.scene.addChild( trident );
        }
        
        private function clickExportBtnHandler( event:MouseEvent ):void
        {
            var exporter    :ThreeJSExporter = new ThreeJSExporter();
            exporter.addEventListener( ExporterEvent.COMPLETE, exporterCompleteHandler );
            exporter.export( view.scene);
        }
        
        private function exporterCompleteHandler( event:ExporterEvent ):void
        {
            trace(event.data);
            System.setClipboard( event.data );
            
            feedbackTf.text     = "HTML copied to clipboard.";
        }
        
        
        private function keyDown(e:KeyboardEvent):void
        {
            lastKey = e.keyCode;
            keyIsDown = true;
        }
        private function keyUp(e:KeyboardEvent):void
        {
            keyIsDown = false;
        }
        
        private function onEnterFrame(e:Event):void
        {
            var moveDist    :uint = 20;
            var rotateStep    :uint = 5;
            
            if(keyIsDown){
                // if the key is still pressed, just keep on moving
                switch(lastKey){
                    case KEY_W            : cam.moveUp(moveDist); break;
                    case KEY_S            : cam.moveDown(moveDist); break;
                    case KEY_A            : cam.moveLeft(moveDist); break;
                    case KEY_D            : cam.moveRight(moveDist); break;
                    case Keyboard.UP    : cam.moveForward(moveDist); break;
                    case Keyboard.DOWN    : cam.moveBackward(moveDist); break;
                    case Keyboard.LEFT    : cam.yaw( -rotateStep ); break;
                    case Keyboard.RIGHT    : cam.yaw( rotateStep ); break;
                }
            }
            // render the view
            view.render();
        }
    }
}
