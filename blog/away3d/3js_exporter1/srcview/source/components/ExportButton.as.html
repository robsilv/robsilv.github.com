package components
{
    import flash.display.MovieClip;
    import flash.display.Sprite;
    import flash.events.MouseEvent;
    import flash.text.TextField;
    
    public class ExportButton extends MovieClip
    {
        public function ExportButton()
        {
            super();
            
            var w        :uint = 108;
            var h        :uint = 24;
            var colour    :uint = 0xAAAAAA;
            
            graphics.beginFill(colour);
            graphics.drawRect(0,0, w, h);
            graphics.endFill();
            
            var tf    :TextField = new TextField();
            tf.text = "EXPORT TO 3JS";
            tf.selectable = false;
            addChild(tf);
            
            tf.x    = 5;
            tf.y    = 2;
            
            var fg    :Sprite = new Sprite();
            fg.graphics.beginFill(colour, 0);
            fg.graphics.drawRect(0,0, w, h);
            fg.graphics.endFill();
            
            addChild(fg);
            
            enable();
        }
        
        public function enable():void
        {
            buttonMode = true;
            useHandCursor = true;
            alpha = 1;
        }
        public function disable():void
        {
            buttonMode = false;
            useHandCursor = false;
            alpha = 0.5;
        }
    }
}
